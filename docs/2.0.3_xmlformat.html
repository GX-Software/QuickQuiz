兼容题库文件格式

兼容题库文件可被任何版本QuickQuiz读取，也可供QuickQuiz-M使用，本质为无BOM头的XML文件，完整格式如下：
红色标出的标签或属性必须存在，否则将视为题库损坏
蓝色标出的内容为若父标签存在，则必须存在的内容，否则父标签包含内容不予读取
<!--标签、属性名称区分大小写-->
<!--推荐编码方式为UTF-8，对于简体中文系统，也可用GBK编码（用于兼容Windows 95，但不推荐）-->
<?xml version="1.0" encoding="UTF-8" ?>
<library>
    <!--题库头，区分大小写-->
    <key>QuickQuiz</key>
    <!--版本号采用四段数字形式，如2.0.0.0，建议使用生成题库所用的应用程序版本号，目前此段不做检查-->
    <version>版本号</version>
    <title>题库名称</title>
    <!--默认封面信息，各属性分别为宽度、高度、背景色、文字颜色、图像品质-->
    <!--若无默认封面，可以无此项-->
    <cover width="300" height="300" clrbk="#FFFFFF" clrtxt="#000000" quality="85">封面文字</cover>

    <!--自定义类型列表，若无自定义类型，可以无此项-->
    <!--由于历史原因，index属性允许缺失，缺失后，将以文档顺序对类型进行编号，但强烈建议包含index属性-->
    <!--自定义类型序号（index属性）必须从1开始并连续增加，否则将以文档顺序对类型进行编号-->
    <!--当包含index属性时，列表的顺序允许打乱，建议按照index序号顺序进行排列-->
    <!--qnum属性代表题号，默认为0-->
    <cust type="基础类型" index="1" qnum="1">自定义类型1名称</cust>
    <!--...-->
    <cust type="基础类型" index="n">自定义类型n名称</cust>
    <!--基础类型名称参考以下各题目-->

    <!--以下题目答案：t为正确、f为错误，不区分大小写-->
    <!--cust属性对应自定义类型列表中index属性，若无cust属性，或cust属性为0，视为默认类型-->
    <!--各题目的顺序允许打乱，但为了便于保存收藏夹文件，建议按固定顺序保存-->
    <!--每题必须包含desc标签，即使内容为空也不能忽略；res标签可忽略-->

    <!--单选题-->
    <qstn type="s-choise" cust="0">
        <desc>题目描述</desc>
        <res>题目解析</res>
        <!--支持bmp、jpeg、png格式图像，题干和解析中均可插入不超过10张图片-->
        <!--图像内容按DataUrl方式存储，不做压缩-->
        <img title="题干图像1标题" index="1" loc="desc">题干图像1内容</img>
        <img title="题干图像2标题" index="2" loc="desc">题干图像2内容</img>
        <img title="解析图像1标题" index="1" loc="res">解析图像1内容</img>
        <!--单选题与多选题应至少包含一个选项，否则认为题目无效-->
        <opt ans="t">选项1</opt>
        <opt ans="f">选项2</opt>
        <opt ans="f">选项3</opt>
    </qstn>

    <!--多选题-->
    <qstn type="m-choise" cust="0">
        <desc>题目描述</desc>
        <res>题目解析</res>
        <opt ans="t">选项1</opt>
        <opt ans="f">选项2</opt>
        <opt ans="t">选项3</opt>
    </qstn>

    <!--判断题-->
    <qstn type="judge" cust="0">
        <desc>题目描述</desc>
        <res>题目解析</res>
        <ans>t</ans>
    </qstn>

    <!--填空题-->
    <qstn type="blank" cust="0">
        <desc>题目描述</desc>
        <res>题目解析</res>
        <!--填空题应至少包含一个空白，否则认为题目无效-->
        <!--空白顺序允许打乱，但为了便于读取，建议按照实际顺序排列-->
        <blk beg="空白起始位置" end="空白结束位置">答案1</blk>
        <blk beg="空白起始位置" end="空白结束位置">答案2</blk>
    </qstn>

    <!--简答题-->
    <qstn type="text" cust="0">
        <desc>题目描述</desc>
        <res>题目解析</res>
        <ans>答案</ans>
    </qstn>

    <!--题目组-->
    <qstn type="group" cust="0" qnum="2">
        <!--题目组描述与题干允许缺失，但建议包含-->
        <type>题目组描述</type>
        <desc>题目组题干</desc>
        <!--这里假设题目组中有一道单选题-->
        <qstn type="s-choise" cust="0">
            <!--除填空题外，题目组中的子题目允许缺失题目描述与解析，但建议包含-->
            <desc>题目描述</desc>
            <res>题目解析</res>
            <opt ans="t">选项1</opt>
            <opt ans="f">选项2</opt>
            <opt ans="f">选项3</opt>
        </qstn>
    </qstn>

    <!--自定义题目（例如自定义简答题）-->
    <!--cust属性对应自定义类型必须与type属性相同，否则不能读取-->
    <qstn type="text" cust="1">
        <desc>题目描述</desc>
        <res>题目解析</res>
        <ans>答案</ans>
    </qstn>

    <!--1.x版本生成的兼容题库中，题目组有单独的标签“grp”-->
    <!--“grp”题目组与“qstn”题目组仅标签名不同-->
    <!--本项仅作为兼容用途，在未来的版本中将完全淘汰这个标签-->
    <grp qnum="2">
        <!--题目组描述与题干允许缺失，但建议包含-->
        <type>题目组描述</type>
        <desc>题目组题干</desc>
        <!--这里假设题目组中有一道单选题-->
        <qstn type="s-choise" cust="0">
            <!--除填空题外，题目组中的子题目允许缺失题目描述与解析，但建议包含-->
            <desc>题目描述</desc>
            <res>题目解析</res>
            <opt ans="t">选项1</opt>
            <opt ans="f">选项2</opt>
            <opt ans="f">选项3</opt>
        </qstn>
    </grp>
</library>
兼容题库文件中部分内容发生意外变动，可能依然能够正常读取，但读取后可能产生未知错误。
若无跨平台或与旧版本兼容等目的，仍然推荐将题库保存为标准题库文件。
  • 自定义类型序号
每个自定义类型均有一个自定义类型序号，从1开始依次连续编号，无上限，用来唯一标识一个自定义类型。自定义类型序号不是固定的，在QuickQuiz内部，自定义类型序号在读入题库文件、添加或删除自定义类型后将自动重新分配。
